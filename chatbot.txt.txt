// 1. find() now returns null when no KB match found
function find(q){
  var ql=q.toLowerCase().trim(),padded=' '+ql+' ';
  var bestKey=null,bs=0;
  for(var k in D){
    var v=D[k],s=0;
    for(var i=0;i<v.kw.length;i++){if(ql.indexOf(v.kw[i])!==-1)s+=v.kw[i].length;}
    if(s>bs){bs=s;bestKey=k;}
  }
  if(!bestKey||bs===0)return null;
  // Always respond to conversational intents (greetings, reactions, thanks)
  if(bestKey==='hello'||bestKey==='reaction'||bestKey==='thank')return D[bestKey].a;
  // Detect general knowledge question patterns (e.g. "what is X", "who is X", "tell me about X")
  var gp=['what is ','what are ','what was ','what were ','tell me about ','explain ','define ',
          'describe ','who is ','who was ','who were ','where is ','where was ','why is ',
          'why was ','why does ','when is ','when was ','when did ',
          'what happened ','who invented ','how does ','how did '];
  var isGenQ=gp.some(function(p){return ql.indexOf(p)===0;});
  if(isGenQ){
    // Require school context (whole-word match using padded string)
    var sc=['isas','school','seeb','oman','muscat','teacher','teachers','principal',
            'student','class','your','our','here',
            'admission','fee','fees','bus','uniform','faculty','syllabus','curriculum','smc'];
    var hasCtx=sc.some(function(w){return padded.indexOf(' '+w+' ')!==-1;});
    if(!hasCtx)return null;
  }
  return D[bestKey].a;
}

function addMsg(h,u){var d=document.createElement('div');d.className='msg '+(u?'user':'bot');d.innerHTML=h;chat.appendChild(d);chat.scrollTop=chat.scrollHeight;}

// 2. New: async Wikipedia search using free Wikipedia REST API (no key needed)
async function searchWikipedia(q){
  try{
    var sr=await fetch('https://en.wikipedia.org/w/api.php?action=opensearch&search='+encodeURIComponent(q)+'&limit=1&format=json&origin=*');
    var sd=await sr.json();
    if(!sd[1]||!sd[1].length)return null;
    var title=sd[1][0],url=sd[3][0];
    var pr=await fetch('https://en.wikipedia.org/api/rest_v1/page/summary/'+encodeURIComponent(title));
    var pd=await pr.json();
    if(pd.extract){
      var ex=pd.extract.length>280?pd.extract.substring(0,280)+'...':pd.extract;
      return '<span style="font-size:.65rem;background:rgba(99,102,241,.18);color:#a5b4fc;padding:2px 9px;border-radius:10px;display:inline-block;margin-bottom:4px">&#127760; Wikipedia</span><br><strong>'+title+'</strong><br><span style="color:#94a3b8">'+ex+'</span><br><a href="'+url+'" target="_blank" style="color:#93c5fd;font-size:.75rem">Read full article &#8599;</a>';
    }
    return null;
  }catch(e){return null;}
}

// 3. reply() is now async — tries KB first, falls back to Wikipedia
async function reply(q){
  typing.style.display='flex';
  chat.appendChild(typing);
  chat.scrollTop=chat.scrollHeight;
  var ans=find(q);
  if(ans){
    setTimeout(function(){typing.style.display='none';addMsg(ans,false);},700);
  }else{
    var wiki=await searchWikipedia(q);
    typing.style.display='none';
    addMsg(wiki||"Not sure about that &#8212; try <strong>admissions</strong>, <strong>teachers</strong>, <strong>sports</strong>, <strong>fees</strong>, or <strong>contact</strong>!",false);
  }
}
